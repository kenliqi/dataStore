CREATE KEYSPACE DEV
WITH REPLICATION = { 'class' : 'SimpleStrategy', 'replication_factor' : 1 };

use DEV;

CREATE TABLE ib_data_Person (
  name text PRIMARY KEY,
  school text,
  age int
);


CREATE TABLE ib_data_TickerQuote (
  ticker varchar,
  date timestamp,
  open double,
  close double,
  high double,
  low double,
  volume double,
  PRIMARY KEY (ticker,date)
) with clustering order by (date desc);
